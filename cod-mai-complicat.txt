void setup(){
  // SeteazÄƒ pin-ul buzzer-ului
  pinMode(13, OUTPUT);
  lcd.begin(16,2);
  
  pinMode(Blue, OUTPUT);  //Blue
  pinMode(Green, OUTPUT);  //Green
  pinMode(Red, OUTPUT); //Red
  pinMode(Buzzer, OUTPUT); //Buzzer

  pinMode(Button, INPUT);   //Button
}


void loop() {

delay(1000);
